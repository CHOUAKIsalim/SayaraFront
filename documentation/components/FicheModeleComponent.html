})
export class FicheModeleComponent implements OnInit {
  public modele: ModeleDetail;
  constructor(private dialogReference: MatDialogRef<FicheModeleComponent>) { }

  /**
   * Ne fait rien
   */

  }

    this.dialogReference.close();
  }

}
